    ///
/// The LOGIN mixin is a basic implementation of a login module that includes
/// an email input, password input, submit button, forgot password link, and
/// remember me checkbox.
///

@mixin login($background: #EEE) {
    background-color: $background;
    margin-left: auto;
    margin-right: auto;
    max-width: 600px;
    padding: 3rem;

    &__field {
        margin-bottom: 1.5rem;
    }

    &__input {
        &:focus {
            box-shadow: 0 0 5px rgba(114, 164, 0, 0.7);
        }
    }

    &__submit {
        text-align: center;
    }

    &__actions {
        text-align: center;
    }

    @include breakpoint(md) {
        &__actions {
            align-items: center;
            display: flex;
            justify-content: space-between;
            flex-flow: row wrap;
        }

        &__submit,
        &__rememberMe,
        &__forgotPasswordLink {
            margin: 0;
            padding: 0;
        }

        &__rememberMe {
            margin-left: 1em;
        }

        &__forgotPasswordLink {
            display: block;
            margin-left: auto;
            text-align: right;
        }
    }
}

@if $use-boilerplate-classes {
    .login {
        @include login;
    }
}
